Object-Oriented Programming (OOP) is a programming paradigm based on the concept of "objects" that contain data and methods. It makes code reusable, modular, and easier to manage.

Example: OOPS in C++

#include <iostream>
using namespace std;

// Base class
class Animal {
public:
    void eat() {
        cout << "This animal eats food" << endl;
    }
};

// Derived class
class Dog : public Animal {
public:
    void sound() {
        cout << "Dog barks" << endl;
    }
};

int main() {
    Dog d1;
    d1.eat();    // Inherited method
    d1.sound();  // Dog's own method
    return 0;
}

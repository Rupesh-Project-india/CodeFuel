Introduction to Swift

Swift is Apple's modern programming language for iOS, macOS, and watchOS development.

Basic Syntax

import Foundation

print("Hello, World!")

var name: String = "Rupesh"
let age: Int = 20

Conditional Statements

let age = 20

if age >= 18 {
    print("You are an adult")
} else {
    print("You are a minor")
}

Loops

// For loop
for i in 1...5 {
    print("Number \(i)")
}

// While loop
var count = 1
while count <= 5 {
    print("Count \(count)")
    count += 1
}

Functions

func greet(name: String) {
    print("Hello, \(name)!")
}

greet(name: "Rupesh")

Conclusion

Swift is powerful, safe, and perfect for Apple ecosystem development.
